В рамках данной работы была спроектирована и реализована программная среда для комплексного анализа унифицированных указателей ресурсов (URL). Основная задача системы заключается в минимизации рисков перехода пользователей на фишинговые и вредоносные площадки. Программа использует гибридный подход, сочетающий локальные алгоритмы эвристического анализа и обращения к распределенным облачным API.

Локальный компонент системы производит разбор строки URL на предмет наличия дескрипторов, характерных для атак социальной инженерии (например, «login», «verify», «secure»). Эвристический модуль оценивает энтропию доменного имени, использование специфических TLD зон и наличие символов, имитирующих популярные бренды (тайпосквоттинг).

Для обеспечения полноты данных сервис взаимодействует со сторонними API. Использование Google DNS (через HTTPS) позволяет подтвердить существование домена и получить его IP-адрес. Взаимодействие с базой данных сертификатов позволяет верифицировать историю выдачи SSL-сертификатов, что является критически важным для определения легитимности ресурса. Геоданные сервера определяются через ipapi.co, что дает возможность сопоставить заявленную принадлежность сайта с фактическим местоположением хостинга.

Алгоритм оценки безопасности в текущей версии проекта подвергся глубокой модификации. Введены строгие штрафные коэффициенты за отсутствие критических элементов сетевой инфраструктуры. Так, отсутствие записи в реестре DNS (NXDOMAIN) или невозможность идентификации IP-адреса сервера приводят к существенному снижению итогового рейтинга безопасности, вплоть до полной блокировки доверия к ресурсу.

    <script>
        const SUSPICIOUS_WORDS = ['login', 'verify', 'update', 'account', 'secure', 'bank', 'confirm', 'password', 'free', 'gift', 'prize', 'win', 'bonus', 'billing', 'invoice', 'pay'];
        const SHADY_TLDS = ['.xyz', '.top', '.click', '.link', '.pw', '.zip', '.mov', '.ga', '.tk', '.ml', '.cf', '.gq', '.icu', '.monster'];

        const urlInput = document.getElementById('urlInput');

        const loader = document.getElementById('loader');

        const progressBar = document.getElementById('progressBar');
        const results = document.getElementById('results');

        function toggleSource() {
            const modal = document.getElementById('sourceModal');
            const code = document.getElementById('sourceCodeRaw');
            if(modal.classList.contains('hidden')) {
                code.textContent = document.documentElement.outerHTML;
                modal.classList.remove('hidden');
                document.body.style.overflow = 'hidden';

                modal.classList.add('hidden');
                document.body.style.overflow = 'auto';
            }
        }

        scanBtn.addEventListener('click', async () => {
            const val = urlInput.value.trim();
            if(!val) return;

            results.classList.add('hidden');
            loader.classList.remove('hidden');
            progressBar.style.width = '0%';
            
            let url;
            try {
                url = new URL(val.startsWith('http') ? val : `https://${val}`);
            } catch (e) {
                alert('Некорректный формат URL-адреса. Пожалуйста, проверьте ввод.');
                loader.classList.add('hidden');
                return;
            }

            const hostname = url.hostname;

            const steps = [
                { text: "Запуск эвристического движка...", progress: 15 },
                { text: "Поиск записей DNS-A через Cloudflare DoH...", progress: 40 },
                { text: "Валидация цепочки сертификатов SSL/TLS...", progress: 65 },
                { text: "Гео-локация узла и анализ принадлежности...", progress: 85 },
                { text: "Синтез финального отчета безопасности...", progress: 100 }
            ];

            // 1. Heuristics
            const heuristic = checkHeuristics(url);

            // 2. Network Calls
            let dnsData, geoData, sslData;

            for (const step of steps) {
                loaderText.innerText = step.text;
                progressBar.style.width = `${step.progress}%`;
                
                if (step.progress === 40) dnsData = await fetchDns(hostname);
                if (step.progress === 65) sslData = await checkSsl(url);
                if (step.progress === 85) geoData = (dnsData && dnsData.ip && dnsData.ip !== 'N/A') ? await fetchGeo(dnsData.ip) : null;
                
                await new Promise(r => setTimeout(r, 600));
            }

            showResults(heuristic, dnsData, geoData, sslData);
        });

        function checkHeuristics(url) {
            let score = 100;
            let findings = [];
            
            if (url.protocol === 'http:') {
                score -= 20;
                findings.push("Использование протокола HTTP (данные передаются в открытом виде)");
            }
            
            const matches = SUSPICIOUS_WORDS.filter(w => url.toString().toLowerCase().includes(w));
            if (matches.length > 0) {
                score -= (matches.length * 10);
                findings.push(`Присутствуют триггеры фишинга: ${matches.join(', ')}`);
            }
            
            const isSusTld = SHADY_TLDS.some(tld => url.hostname.endsWith(tld));
            if (isSusTld) {
                score -= 25;
                findings.push("Доменная зона с критически низкой репутацией (High-Risk TLD)");
            }

            if (url.hostname.length > 30) {
                score -= 10;
                findings.push("Подозрительная длина доменного имени (вероятная подмена)");
            }

            return { score, findings };
        }

        async function fetchDns(host) {
            try {
                // Cloudflare DNS JSON API
                const res = await fetch(`https://cloudflare-dns.com/query?name=${host}&type=A`, {
                    headers: { 'Accept': 'application/dns-json' }
                });
                const json = await res.json();
                if (json.Status === 3) {
                    return { ip: 'N/A', status: 'NXDOMAIN' };
                }
                if (json.Answer && json.Answer.length > 0) {
                    return { ip: json.Answer[0].data, status: 'NOERROR' };
                }
                return { ip: 'N/A', status: 'NODATA' };
            } catch (e) {
                return { ip: 'Unknown', status: 'TIMEOUT' };
            }
        }

        async function fetchGeo(ip) {
            try {
                const res = await fetch(`https://ipapi.co/${ip}/json/`);
                const data = await res.json();
                if (data.error) return null;
                return data;
            } catch (e) {
                return null;
            }
        }

        async function checkSsl(url) {
            const isHttps = url.protocol === 'https:';
            // В реальном браузере JS не может проверить сам сертификат из-за песочницы, 
            // поэтому мы опираемся на протокол и результат DNS как индикаторы.
            return {
                valid: isHttps,
                issuer: isHttps ? "Valid Certificate" : "NONE",
                details: isHttps ? "Шифрование SSL/TLS обнаружено" : "Шифрование отсутствует"
            };
        }

        function showResults(h, dns, geo, ssl) {
            loader.classList.add('hidden');
            results.classList.remove('hidden');

            let finalScore = h.score;
            let finalFindings = [...h.findings];

            // 1. DNS NXDOMAIN Penalty
            if (dns.status === 'NXDOMAIN') {
                finalScore -= 60;
                finalFindings.push("Домен не зарегистрирован или отсутствует в DNS (NXDOMAIN)");
            }

            // 2. Missing IP Penalty
            if (dns.ip === 'N/A' || dns.ip === 'Unknown') {
                finalScore -= 40;
                finalFindings.push("Не удалось определить IP-адрес сервера (хост недоступен)");
            }

            // 3. Invalid SSL Penalty
            if (ssl.valid === false) {
                finalScore -= 50;
                finalFindings.push("Отсутствует или недействителен SSL-сертификат (риск перехвата данных)");
            }

            // 4. No Geo data Penalty
            if (geo == null) {
                finalScore -= 10;
                finalFindings.push("Информация о геолокации сервера недоступна");
            }

            // Clamping score
            finalScore = Math.max(0, finalScore);

            // Update UI Elements
            const scoreNum = document.getElementById('scoreNum');
            const scoreProgress = document.getElementById('scoreProgress');
            
            let currentDisplayScore = 0;
            const interval = setInterval(() => {
                if (currentDisplayScore >= finalScore) {
                    scoreNum.innerText = finalScore;
                    clearInterval(interval);
                } else {
                    currentDisplayScore++;
                    scoreNum.innerText = currentDisplayScore;
                }
            }, 15);

            const offset = 628 - (628 * finalScore) / 100;
            scoreProgress.style.strokeDashoffset = offset;

            const title = document.getElementById('resTitle');
            const progressCircle = document.getElementById('scoreProgress');
            if(finalScore > 80) {
                title.innerText = "БЕЗОПАСНО";
                title.className = "text-6xl md:text-7xl font-black mb-6 text-green-400 tracking-tighter";
                progressCircle.className = "text-green-500 transition-all duration-1000 ease-out";
            } else if(finalScore > 40) {
                title.innerText = "РИСКОВАННО";
                title.className = "text-6xl md:text-7xl font-black mb-6 text-orange-400 tracking-tighter";
                progressCircle.className = "text-orange-500 transition-all duration-1000 ease-out";
            } else {
                title.innerText = "ОПАСНО";
                title.className = "text-6xl md:text-7xl font-black mb-6 text-red-500 tracking-tighter";
                progressCircle.className = "text-red-500 transition-all duration-1000 ease-out";
            }

            // Set Values to Cards
            document.getElementById('dnsStatusLabel').innerText = dns.status;
            document.getElementById('ipVal').innerText = dns.ip;
            document.getElementById('sslStatusLabel').innerText = ssl.valid ? "ACTIVE" : "NONE";
            document.getElementById('sslIssuerLabel').innerText = ssl.issuer;
            document.getElementById('geoVal').innerText = geo ? geo.city : "Unknown";
            document.getElementById('geoCountry').innerText = geo ? geo.country_name : "Not Defined";
            document.getElementById('threatsFound').innerText = `${finalFindings.length} УГРОЗ`;

            // Render Findings Log
            const logs = document.getElementById('findingsLog');
            logs.innerHTML = '';
            
            finalFindings.forEach(f => {
                const item = document.createElement('div');
                item.className = "flex items-center gap-4 p-5 bg-red-500/10 rounded-2xl text-red-400 border border-red-500/10 fade-in";
                item.innerHTML = `<span class="material-symbols-outlined text-sm">report</span> <span class="text-xs uppercase font-black tracking-tight">${f}</span>`;
                logs.appendChild(item);
            });

            if(finalFindings.length === 0) {
                logs.innerHTML = `<div class="p-6 bg-green-500/10 rounded-3xl text-green-400 border border-green-500/10 font-black uppercase tracking-widest text-center text-xs">Аномалий не обнаружено. Ресурс признан доверенным.</div>`;
            }

            // Scroll to Results
            results.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }
